00025 class Menu_Light extends AbstractView {
00026         /*
00027            Sometimes you want to have a HTML-based menu, and all you need PHP to do is to highlight current page.
00028            It's not that simply always, sometimes there are sub-pages also. This class solves the problem.
00029 
00030            Lets build a structure which adds asterik next to page. First we need to specify that the object to be
00031            coppied around is an asterik. (in your case it might be a class name)
00032 
00033            <?current?>*<?/?>
00034 
00035            Next put self closing tags in form "current_" + page exactly where you want the current to appear
00036 
00037            services <?$current_services?>
00038            consulting <?$current_services_consulting?>
00039            mentoring <?$current_services_consulting?>
00040            products <?$current_products?>
00041            about <?$current_about?><?$current_contact?>
00042 
00043            If you go to page servcies, the asterisk will be placed next to "services". If you are on the page services/mentoring
00044            then asterik will appear next to services AND consulting lines. 
00045 
00046            Line with "about" will have asterik if you are either on 'about' or 'contact' page. That's all you need to know to use
00047            this.
00048 
00049            To add thhis to your application use $this->add('Menu_Light',null,'Menu');
00050          */
00051         function render(){
00052                 $c=$this->template->get('current');
00053                 $this->template->del('current');
00054 
00055                 $parts=explode('_',$this->api->page);
00056 
00057                 $matched=false;
00058                 while($parts){
00059                         $tag=implode('_',$parts);
00060 
00061                         if($this->template->is_set('current_'.$tag)){
00062                                 $this->template->trySet('current_'.$tag,$c);
00063                                 $matched=true;
00064                         }
00065                         array_pop($parts);
00066                 }
00067 
00068                 if(!$matched){
00069                         $this->template->set('current',$c);
00070                 }
00071                 parent::render();
00072         }
